/**

/** 
 * @summary 
 * Clickable blue button. PROUT !!!
 * 
 * @description
 * Accept two other classes :
 * <ul>
 * <li><b>green</b> : green style</li>
 * <li><b>red</b> : red style</li>
 * </ul>
 * 
 * @example
 * <w:button fire="add" fire-arg="27">Edit</w:button>
 * <w:button fire="add:27">Edit</w:button>
 * 
 * @param fire Name of the signal to trigger when user clicks on the button.
 *             To add an argument, put  it in the "fire-arg" attribute
 *             or  add it  tho  this  one separated  by  a colon  (ex:
 *             fire="edit:27").
 * @param fire-arg Argument to fire with the signal.
 */
module.exports = function() {
    var N = this.Node;
    var root = this.root;
    var result = this.result;
    root.name = "a";
    var fire = N.att(root, "fire");
    var fireArg = N.att(root, "fire-arg");
    if (typeof fire === 'string') {
        delete root.attribs.fire;
        delete root.attribs["fire-arg"];
        if (typeof fireArg !== 'string') {
            var pos = fire.indexOf(":");
            if (pos > -1) {
                fireArg = fire.substr(pos + 1);
                fire = fire.substr(0, pos);
            }            
        }
        if (fire) result.init.fire = fire;
        if (fireArg) result.init.fireArg = fireArg;
    }
    N.addClass(root, "wtag-button");
};
