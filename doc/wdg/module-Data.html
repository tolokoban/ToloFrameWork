Module: Data

Data

Define a data binded in the current scope.

Source:
  • data/compile-data.js, line 1

Example

<!--
 This will produce this data binding :
 parent.$data = {
   "context": [
     {"sens":["Vue", "Ouïe", "Odorat", "Goût", "Toucher"]},
     []
   ]
 }
-->
<w:Data name="context">
 <map>
   <item key="sens">
     <list>
       <item>Vue</item>
       <item>Ouïe</item>
       <item>Odorat</item>
       <item>Goût</item>
       <item>Toucher</item>
     </list>
   </item>
 </map>
</w:Data>
